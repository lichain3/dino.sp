#include <iostream>
using namespace std;

int main()
{

    setlocale(LC_ALL, "Russian");
    int k;
    string sss[5][5] = { {" История"," Химия"," Физика"," Литература"," Физкультура"},
                         {" Физкультура"," История"," Химия"," Физика"," Литература"},
                         {" Литература"," Физкультура"," История"," Химия"," Физика"},
                         {" Физика"," Литература"," Физкультура"," История"," Химия"},
                         {" Химия"," Физика"," Литература"," Физкультура"," История"},};

    cout << "Введите день недели";
    cin >> k;

    for (int i = 0; i < 5; i++)
    {
        cout << "\n" << i + 1 << sss[k - 1][i];
    }
}
