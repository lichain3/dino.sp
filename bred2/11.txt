#include <iostream>
#include <windows.h>

using namespace std;

int main()
{
    setlocale(LC_ALL, "Russian");

    int x, k, m=9;
    cout<<"Ведите кол-во строк: ";
    cin>>x;


    cout<<"Ведите кол-во столбцов: ";
    cin>>k;


    int arr[x][k];
    
    for(int i=0; i<x; i++)
    {
        for(int q=0; q<k; q++)
        {
            arr[i][q]=1+ rand() % m;
        }
    }

    for(int i=0; i<x; i++)
    {
        for(int q=0; q<k; q++)
        {
            if(arr[i][q]>l)
                l=arr[i][q];
            
            if(arr[i][q]<m)
                m=arr[i][q];
        }
    }
    
    cout<<"min: "<<m<<"\nmax: "<<l;
    return 0;
}
