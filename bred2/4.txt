{

	string arr[7] = { "Понидельник", "Вторник","Среда","Четверг","Пятница","Суббота","Воскрисенье" };
	int d = 10, m = 1, g = 2018, c, nid;

	cout << "Введите день";
	cin >> d;
	cout << "Введите месяц";
	cin >> m;
	cout << "Введите год";
	cin >> g;

	if (m < 3)

		m += 10;
		g--;
	}
