#include <iostream>
#include <windows.h>

using namespace std;

int main()
{
    setlocale(LC_ALL, "Russian");
    int x, y;
    cout<<"Введите кол-во чисел: ";
    cin>>x;

    int arr[x];

    for(int i=0; i<x; i++)
    {
        arr[i] = 1 + rand() % 9;

        cout<<arr[i]<<"   ";
    }

    cout<<"Ведите сколько чисел надо посчитать: ";
    cin>>y;

            x=0;

    for(int i=0; i<y; i++)
            x+=arr[i];
    cout<<"Ответ:  "<<x;
    return 0;
}
