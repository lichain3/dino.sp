#include <iostream>
#include <windows.h>

using namespace std;

int main()
{
    setlocale(LC_ALL, "Russian");
    int arr[10], x, y;

    for(int i=0; i<10; i++)
    {
        arr[i] = 1 + rand() % 9;

        cout<<arr[i]<<"   ";
    }

    cout << "\n¬ведите числа:\n";

    do{
        cin>>x;
        for(int i=0; i<10; i++)
        {
            if(x==arr[i])
                y=1;
        }

        if(y==1)
            cout<<"  -  Ёто число есть в массиве\n";
        else
            cout<<"  -  Ётого числа нет в массиве\n";

        y=0;
    }while(x!= 999);

    return 0;
}
